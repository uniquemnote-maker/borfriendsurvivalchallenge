<script>
    const notification = document.getElementById('notification');
    const senderName = document.getElementById('sender-name');
    const msgContent = document.getElementById('msg-content');
    const targetIcon = document.getElementById('target-icon');
    const scoreDisplay = document.getElementById('score-display');
    const overlay = document.getElementById('overlay');
    const startBtn = document.getElementById('start-btn');
    let score = 0, gameActive = false, timer = null, currentSpeed = 1500, isTarget = true;
    const girlfriendMsgs = ["想我嗎？❤️", "這張照片好看嗎？", "肚子餓了啦...", "愛你喔！", "在幹嘛？", "快理我！"];
    const otherMsgs = [
        { name: "老闆", msg: "報表交了沒？" },
        { name: "外送員", msg: "餐點已送達門口" },
        { name: "詐騙", msg: "您的包裹地址錯誤" },
        { name: "廣告", msg: "限時特價 1 折起！" },
        { name: "貓咪", msg: "喵～(要摸摸)" }
    ];
    startBtn.addEventListener('click', (e) => { e.stopPropagation(); startGame(); });
    function startGame() { score = 0; currentSpeed = 1500; gameActive = true; scoreDisplay.innerText = score; overlay.style.display = 'none'; spawnNotification(); }
    function spawnNotification() {
        if (!gameActive) return;
        isTarget = Math.random() > 0.35; 
        if (isTarget) {
            senderName.innerText = "親愛的女友";
            msgContent.innerText = girlfriendMsgs[Math.floor(Math.random() * girlfriendMsgs.length)];
            targetIcon.className = "notif-icon icon-target";
        } else {
            const other = otherMsgs[Math.floor(Math.random() * otherMsgs.length)];
            senderName.innerText = other.name;
            msgContent.innerText = other.msg;
            targetIcon.className = "notif-icon icon-other";
        }
        notification.classList.add('show');
        timer = setTimeout(() => { if (isTarget) gameOver("太慢了！漏掉女友訊息！"); else nextTurn(); }, currentSpeed);
    }
    function handleAction(type, event) {
        if (!gameActive || !notification.classList.contains('show')) return;
        event.preventDefault();
        const x = event.changedTouches ? event.changedTouches[0].clientX : event.clientX;
        const y = event.changedTouches ? event.changedTouches[0].clientY : event.clientY;
        if (isTarget) {
            if (type === 'like') { clearTimeout(timer); score++; scoreDisplay.innerText = score; createHeart(x, y); if (currentSpeed > 450) currentSpeed -= 70; nextTurn(); }
            else { gameOver("不准點回覆，只能點讚！"); }
        } else { gameOver(`你竟然點了「${senderName.innerText}」的讚？！`); }
    }
    const lBtn = document.getElementById('like-btn'), rBtn = document.getElementById('reply-btn');
    lBtn.addEventListener('touchstart', (e) => handleAction('like', e));
    lBtn.addEventListener('click', (e) => handleAction('like', e));
    rBtn.addEventListener('touchstart', (e) => handleAction('reply', e));
    rBtn.addEventListener('click', (e) => handleAction('reply', e));
    function nextTurn() { notification.classList.remove('show'); setTimeout(spawnNotification, 500); }
    function gameOver(reason) { gameActive = false; notification.classList.remove('show'); overlay.style.display = 'flex'; document.getElementById('overlay-title').innerText = "💀 訓練失敗"; document.getElementById('overlay-desc').innerHTML = `${reason}<br><br>得分：${score}`; }
    function createHeart(x, y) { const heart = document.createElement('div'); heart.innerHTML = '❤️'; heart.className = 'heart'; heart.style.left = (x - 22) + 'px'; heart.style.top = (y - 22) + 'px'; document.body.appendChild(heart); setTimeout(() => heart.remove(), 800); }
</script>
